


function int iterativeFact(num : int, incr : int) { 
	result : int = 1.

	while ({num != 1}) { // Relational operation
		result = ((result * num) * incr). // Multiple arithmetic operations
		num = (num - 1).
	}

	return result.
}

function none main() {
	result : int = call(iterativeFact, 5, 1).

	output("Result factorial: ").
	output(result).
}

call_main(main).
